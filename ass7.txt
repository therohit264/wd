    <h1>Dynamic Web Page using DOM</h1>

    <button onclick="changeText()">Change Text</button>
    <button onclick="addItem()">Add List Item</button>

    <div id="content">Hello! Click the button to change me.</div>

    <ul id="list"></ul>

    <script>
        function changeText() {
            document.getElementById("content").innerHTML = "Text has been changed dynamically!";
        }

        function addItem() {
            let li = document.createElement("li");
            li.textContent = "New Item added!";
            document.getElementById("list").appendChild(li);
        }
    </script>
